FROM students s
JOIN `groups` g ON s.group_id = g.id
JOIN  books b  ON s.id = b.taken_by_student_id
JOIN marks m ON s.id = m.student_id
JOIN lessons l ON m.lesson_id = l.id
JOIN subjects s2 ON s2.id = l.subject_id
